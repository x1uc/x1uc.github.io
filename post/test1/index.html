import java.io.*;  
  
public class Main {  
    static int[] brr = new int[100005];  
    static int[] arr = new int[100005];  
    static int n;  
  
    public static void merge_sort(int l, int r) {  
        if (l == r) {  
            brr[l] = arr[l];  
            return;  
        }  
        int mid = l + r >> 1;  
        int left = l;  
        int right = mid + 1;  
        int pos = l;  
        merge_sort(l, mid);  
        merge_sort(mid + 1, r);  
        while (pos <= r) {  
            if (left > mid) {  
                brr[pos++] = arr[right++];  
            } else if (right > r) {  
                brr[pos++] = arr[left++];  
            } else {  
                if (arr[left] > arr[right]) {  
                    brr[pos++] = arr[right++];  
                } else {  
                    brr[pos++] = arr[left++];  
                }  
            }  
        }  
        for (int i = l; i <= r; i++) {  
            arr[i] = brr[i];  
        }  
    }  
  
    public static void main(String[] args) throws IOException {  
        StreamTokenizer streamTokenizer = new StreamTokenizer(new BufferedReader(new InputStreamReader(System.in)));  
        PrintWriter printWriter = new PrintWriter(new OutputStreamWriter(System.out));  
        streamTokenizer.nextToken();  
        n = (int) streamTokenizer.nval;  
        for (int i = 0; i < n; i++) {  
            streamTokenizer.nextToken();  
            arr[i] = (int) streamTokenizer.nval;  
        }  
        merge_sort(0, n - 1);  
        for (int i = 0; i < n; i++) {  
            printWriter.print(brr[i] + " ");  
        }  
        printWriter.flush();  
        printWriter.close();  
    }  
}

import java.io.*;  
import java.util.Arrays;  
import java.util.LinkedList;  
import java.util.Queue;  
import java.util.Random;  
  
public class Main {  
    static int[] arr = new int[100005];  
    static int n;  
  
    public static void quick_sort(int l, int r) {  
        if (l >= r)  
            return;  
        int pivotIndex = partition(l, r);  
        quick_sort(l, pivotIndex - 1);  
        quick_sort(pivotIndex + 1, r);  
    }  
  
    private static int partition(int l, int r) {  
        int pivot = arr[r];  
        int i = l - 1;  
        for (int j = l; j < r; j++) {  
            if (arr[j] <= pivot) {  
                swap(++i, j);  
            }  
        }  
        swap(i + 1, r);  
        return i + 1;  
    }  
  
    private static void swap(int i, int j) {  
        int temp = arr[i];  
        arr[i] = arr[j];  
        arr[j] = temp;  
    }  
  
    public static void main(String[] args) throws IOException {  
        StreamTokenizer streamTokenizer = new StreamTokenizer(new BufferedReader(new InputStreamReader(System.in)));  
        PrintWriter printWriter = new PrintWriter(new OutputStreamWriter(System.out));  
        streamTokenizer.nextToken();  
        n = (int) streamTokenizer.nval;  
        for (int i = 0; i < n; i++) {  
            streamTokenizer.nextToken();  
            arr[i] = (int) streamTokenizer.nval;  
        }  
        quick_sort(0, n - 1);  
        for (int i = 0; i < n; i++) {  
            printWriter.print(arr[i] + " ");  
        }  
        printWriter.flush();  
        printWriter.close();  
    }  
}
